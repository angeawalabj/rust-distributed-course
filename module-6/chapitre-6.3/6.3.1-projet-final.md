## Étape 2 — Crate protocole (crates/protocole/src/lib.rs)

```rust
// Types partagés entre tous les composants
use serde::{Deserialize, Serialize};
use uuid::Uuid;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RequêteKV {
    pub id: Uuid,          // Idempotence
    pub opération: OpKV,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum OpKV {
    Get { clé: String },
    Set { clé: String, valeur: Vec<u8>, ttl_secondes: Option<u64> },
    Delete { clé: String },
    ListePréfixe { préfixe: String, limite: u32 },
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RéponseKV {
    pub id: Uuid,
    pub résultat: RésultatKV,
    pub nœud_source: u32,
    pub durée_µs: u64,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum RésultatKV {
    Valeur(Vec<u8>),
    Clés(Vec<String>),
    Ok,
    Introuvable,
    Erreur { code: String, message: String },
}

/// Message interne Raft
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum MessageRaft {
    Vote { terme: u64, candidat: u32, dernier_index: u64, dernier_terme: u64 },
    VoteOk { terme: u64, accordé: bool },
    AppendEntries { terme: u64, leader: u32, entrées: Vec<EntréeRaft>, commit: u64 },
    AppendOk { terme: u64, succès: bool },
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct EntréeRaft {
    pub terme: u64,
    pub index: u64,
    pub données: Vec<u8>,  // OpKV sérialisée
}
```

## Étape 3 — Point d'entrée principal (crates/nœud/src/main.rs)

```rust
// crates/nœud/src/main.rs
use std::sync::Arc;
use axum::{middleware, Router, routing::{delete, get, post, put}};

use distribkv_api::{handlers, middleware_jwt};
use distribkv_consensus::MoteurRaft;
use distribkv_observabilite::{Métriques, initialiser_télémétrie, middleware_métriques};
use distribkv_plugins::MoteurPlugins;
use distribkv_résilience::CircuitBreaker;
use distribkv_stockage::KVShardé;

#[derive(Clone)]
pub struct ÉtatApp {
    pub kv: Arc<KVShardé>,
    pub raft: Arc<MoteurRaft>,
    pub plugins: Arc<MoteurPlugins>,
    pub métriques: Arc<Métriques>,
    pub cb_réplication: Arc<CircuitBreaker>,
}

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let config = Config::depuis_env()?;

    // Télémétrie
    initialiser_télémétrie(&config.nom_service, &config.otlp_endpoint)?;

    // Composants
    let kv = Arc::new(KVShardé::nouveau(&config.données_dir, config.nb_shards)?);
    let plugins = Arc::new(MoteurPlugins::nouveau()?);
    let métriques = Arc::new(Métriques::nouvelles(&prometheus::default_registry()));

    // Charger les plugins WASM
    plugins.charger("filtre-défaut", &config.plugins_dir.join("filtre-défaut.wasm")).await?;
    tokio::spawn(distribkv_plugins::surveiller_plugins(
        plugins.clone(),
        config.plugins_dir.clone(),
    ));

    // Raft
    let raft = Arc::new(MoteurRaft::nouveau(
        config.nœud_id,
        config.pairs.clone(),
        kv.clone(),
    ));
    tokio::spawn(raft.clone().boucle_principale());

    let cb = Arc::new(CircuitBreaker::nouveau(
        "réplication-raft",
        Default::default(),
    ));

    let état = ÉtatApp { kv, raft, plugins, métriques: métriques.clone(), cb_réplication: cb };

    let app = Router::new()
        // API KV publique
        .route("/kv/:clé", get(handlers::get).put(handlers::set).delete(handlers::delete))
        .route("/kv",      get(handlers::liste))
        // Admin (authentifié)
        .nest("/admin", Router::new()
            .route("/cluster", get(handlers::état_cluster))
            .route("/plugins",  post(handlers::charger_plugin))
            .layer(middleware::from_fn_with_state(
                état.clone(),
                middleware_jwt::vérifier,
            ))
        )
        // Interne Raft
        .route("/raft/vote",   post(handlers::raft_vote))
        .route("/raft/append", post(handlers::raft_append))
        // Observabilité
        .route("/metrics",    get(distribkv_observabilite::handler_metrics))
        .route("/santé",      get(handlers::santé))
        // Middlewares globaux
        .layer(middleware::from_fn_with_state(
            métriques.clone(),
            middleware_métriques,
        ))
        .layer(tower_http::trace::TraceLayer::new_for_http())
        .with_state(état);

    let addr = format!("0.0.0.0:{}", config.port);
    tracing::info!(nœud_id = config.nœud_id, addr = %addr, "DistribKV démarré");

    axum::serve(
        tokio::net::TcpListener::bind(&addr).await?,
        app,
    ).await?;

    opentelemetry::global::shutdown_tracer_provider();
    Ok(())
}
```

## Étape 4 — Dockerfile multi-stage

```dockerfile
# Dockerfile
FROM rust:1.75-slim AS builder

WORKDIR /app
COPY Cargo.toml Cargo.lock ./
COPY crates/ crates/

# Cache des dépendances
RUN --mount=type=cache,target=/usr/local/cargo/registry \
    --mount=type=cache,target=/app/target \
    cargo build --release --bin nœud

RUN cp target/release/nœud /usr/local/bin/distribkv

# ── Image finale ─────────────────────────────────────────────
FROM debian:bookworm-slim

RUN apt-get update && apt-get install -y ca-certificates && rm -rf /var/lib/apt/lists/*

COPY --from=builder /usr/local/bin/distribkv /usr/local/bin/distribkv

# Plugins WASM
COPY plugins-wasm/filtre-défaut/target/wasm32-wasi/release/filtre_défaut.wasm \
     /app/plugins/filtre-défaut.wasm

EXPOSE 8080

ENV RUST_LOG=info
ENV APP_ENV=production

ENTRYPOINT ["/usr/local/bin/distribkv"]
```

```yaml

apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: distribkv
  namespace: distribkv
spec:
  serviceName: distribkv-headless
  replicas: 3
  selector:
    matchLabels:
      app: distribkv

    metadata:
      labels:
        app: distribkv
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8080"
        prometheus.io/path: "/metrics"
    spec:
      containers:
        - name: distribkv
          image: distribkv:latest
          ports:
            - containerPort: 8080
          env:
            - name: NŒUD_ID
              valueFrom:
                fieldRef:
                  fieldPath: metadata.name  # distribkv-0, distribkv-1, distribkv-2
            - name: PORT
              value: "8080"
            - name: PAIRS
              value: "http://distribkv-1.distribkv-headless:8080,http://distribkv-2.distribkv-headless:8080"
            - name: DONNÉES_DIR
              value: "/data"
            - name: OTLP_ENDPOINT
              value: "http://jaeger:4317"
          resources:
            requests:
              memory: "256Mi"
              cpu: "250m"
            limits:
              memory: "512Mi"
              cpu: "500m"
          readinessProbe:
            httpGet:
              path: /santé
              port: 8080
            initialDelaySeconds: 5
            periodSeconds: 5
          livenessProbe:
            httpGet:
              path: /santé
              port: 8080
            initialDelaySeconds: 15
            periodSeconds: 10
          volumeMounts:
            - name: données
              mountPath: /data
  volumeClaimTemplates:
    - metadata:
        name: données
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: 10Gi
```

## 🏆 Critères de validation du projet final

### Fonctionnel
- [ ] Le cluster élit un leader en < 2 secondes
- [ ] Les écritures sont répliquées sur tous les nœuds
- [ ] La panne d'un nœud n'interrompt pas le service
- [ ] Les plugins WASM filtrent les requêtes correctement
- [ ] L'authentification JWT protège les routes admin

### Performance
- [ ] Throughput > 10 000 ops/sec en écriture sur le cluster
- [ ] Latence P99 < 10ms pour les lectures locales
- [ ] Récupération après panne < 5 secondes

### Observabilité
- [ ] Toutes les requêtes sont tracées (Jaeger)
- [ ] Les métriques sont exposées et scrapées par Prometheus
- [ ] Les dashboards Grafana reflètent l'état en temps réel
- [ ] Les alertes se déclenchent lors des pannes simulées

### Ops
- [ ] Le build Docker est reproductible (multi-stage)
- [ ] Le déploiement K8s utilise des probes liveness/readiness
- [ ] L'HPA scale automatiquement sous charge
- [ ] Les données persistent après le redémarrage d'un pod
